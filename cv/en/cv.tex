\begin{project}{Hi! PARIS Hi!ckathon 2025 (2nd place)}

\end{project}\vspace{-6pt}

\begin{project}{Molecular Graph Captioning (Kaggle competition)}
{\href{https://github.com/maxencedbe/molecular_graph_captioning}{\textcolor{black}{\raisebox{-0.05\height}{\large \faGithub}}}}

    \item Implemented a dedicated graph neural network encoder using PyTorch Geometric, optimized for semantic alignment through BERTScore and BLEU-4 metrics.
\end{project}\vspace{-6pt}

\begin{project}{Streaming Safety Classifier for LLMs --- Capstone Project with DragonLLM (now OVHai LLM)}
{\href{https://github.com/maxencedbe/capstone_dragonllm}{\textcolor{black}{\raisebox{-0.05\height}{\large \faGithub}}}}
    \item Reproduced the Qwen3Guard toxicity classifier: built a token-level labeled dataset of 24K toxic and safe Q\&A samples, synthesized and labeled with Qwen3-30B.
    \item Trained a lightweight per-token safety head on a frozen Qwen backbone to interrupt harmful generations in real time.
\end{project}\vspace{-6pt}

\begin{project}{Data Privacy \& Anonymization --- Project Cassiopée}
{\href{https://github.com/maxencedbe/Cassiopee}{\textcolor{black}{\raisebox{-0.05\height}{\large \faGithub}}}}

\end{project}\vspace{-6pt}

\begin{project}{Automated Parking Optimization}
